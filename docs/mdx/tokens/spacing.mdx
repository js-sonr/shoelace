export function SpacingDemo({ token }) {
  return (
    <div 
      className="spacing-demo" 
      style={{ 
        width: `var(${token})`, 
        height: `var(${token})` 
      }} 
    />
  );
}


# Spacing Tokens

Spacing tokens are used to provide consistent spacing between content in your app.

| Token                   | Value          | Example                                                                                                         |
| ----------------------- | -------------- | --------------------------------------------------------------------------------------------------------------- |
| `--nu-spacing-3x-small` | 0.125rem (2px) | <SpacingDemo token="--nu-spacing-3x-small" /> |
| `--nu-spacing-2x-small` | 0.25rem (4px)  | <SpacingDemo token="--nu-spacing-2x-small" /> |
| `--nu-spacing-x-small`  | 0.5rem (8px)   | <SpacingDemo token="--nu-spacing-x-small" />   |
| `--nu-spacing-small`    | 0.75rem (12px) | <SpacingDemo token="--nu-spacing-small" />       |
| `--nu-spacing-medium`   | 1rem (16px)    | <SpacingDemo token="--nu-spacing-medium" />     |
